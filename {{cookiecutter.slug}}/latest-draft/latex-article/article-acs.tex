  \filename@parse{#1}%
  \InputIfFileExists{#1}{}{%
    \message{LaTeX Warning: File `%
    \filename@base.\ifx\filename@ext\relax tex\else\filename@ext\fi'%
    not found on input line \the\inputlineno}%
  }%
}
